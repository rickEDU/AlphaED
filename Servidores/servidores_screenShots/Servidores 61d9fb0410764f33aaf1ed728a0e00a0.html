sudo apt-get update

sudo cp /etc/vsftpd.conf /etc/vsftpd.conf.original

sudo ufw allow 20/tcp
sudo ufw allow 21/tcp
sudo ufw allow 22/tcp
sudo ufw allow 990/tcp
sudo ufw allow 40000:50000/tcp

#Verificação das portas do firewall
sudo ufw status

#Adição do usuário Alex (exemplo) ao FTP
sudo adduser alex

#Criação de pasta como nome ftp para o usuário Alex (exemplo)
sudo mkdir /home/alex/ftp

#Definição de propriedade no chown
sudo chown nobody:nogroup /home/alex/ftp

#Remoção de permissões para gravar
sudo chmod a-w /home/alex/ftp

#Verificar as permissões
sudo ls -la /home/alex/ftp

#Diretório com os arquivos
sudo mkdir /home/alex/ftp/files

#Permissões para o diretório onde contém os arquivos
sudo chown alex:alex /home/alex/ftp/files

#Exemplo de arquivo para realizar teste
echo "vsftpd sample file" | sudo tee /home/alex/ftp/files/sample.txt

#Aqui deve está o contéudo editado do sudo nano /etc/vsftpd.conf
sudo nano /etc/vsftpd.conf

#Criação e adição de nossos usuários ao arquivo
echo "alex" | sudo tee -a /etc/vsftpd.userlist

#Verifica se o usuário está realmente ativo, logo após este comando a saída deve
#ser o nome do usuário cadastrado
cat /etc/vsftpd.userlist

#Reiniciar o vsftpd
sudo systemctl restart vsftpd

#Criação de certificado SSL
sudo openssl req -x509 -nodes -days 365 -newkey rsa:2048 -keyout /etc/ssl/private/vsftpd.pem -out /etc/ssl/private/vsftpd.pem

      [Unit]
Description=The NGINX HTTP and reverse proxy server
After=syslog.target network-online.target remote-fs.target nss-lookup.target
Wants=network-online.target
        
[Service]
Type=forking
PIDFile=/var/run/nginx.pid
ExecStartPre=/usr/sbin/nginx -t
ExecStart=/usr/sbin/nginx
ExecReload=/usr/sbin/nginx -s reload
ExecStop=/bin/kill -s QUIT $MAINPID
PrivateTmp=true
        
[Install]
WantedBy=multi-user.target

      #Criar uma aplicativo
nano hello.js
#Copiar o código para dentro do hello.js
const http = require('http');

const hostname = 'localhost';
const port = 3000;

const server = http.createServer((req, res) => {
  res.statusCode = 200;
  res.setHeader('Content-Type', 'text/plain');
  res.end('Hello World!\n');
});

server.listen(port, hostname, () => {
  console.log(`Server running at http://${hostname}:${port}/`);
});
#Após salvar e sair, testar o aplicativo
node hello.js
